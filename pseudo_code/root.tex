\section{Introduction}

Algorithm 1 computes the $Reg$ of a Game $G$ for a given value of $b$ and a payoff function $Val$. The construction of $\hat{G}$ depends on computing $W'$ which is primarily computing $cVal$ for each edge $\{e = (u,v) \in G$ \vert $u \in V_\exists \}$ such that Eve plays an alternate strategy $\sigma'$ and $\tau$ for Adam that maximizes the payoff for any of the alternative strategy $\sigma'$ except the original strategy $\sigma$. \\

We then use values($b \in W'$) to construct $G^b$ with the new weight function $\hat{w} = w(e) - b$ to construct $\hat{G}$. On this game it is sufficient to play a memoryless/positional strategies for either players to ensure an antagonistic value of at least(resp. at most) $aVal(\hat{G})$. From \textbf{Section 3 Claim 1.} we conclude that $Reg(G) = -aVal(\hat{G})$. Algorithm 4 initially computes the memoryless strategy for Eve and Adam where Eve looks for the edge with the highest weight and Adam plays adversarially looking for the edge with the minimum weight, to compute the $aVal$ of the game $\hat{G}$. We then can compute $\sigma$ and $\tau$ for the game $G$ as there is an one-to-one correspondence with strategies $\hat{\sigma}$ and $\hat{\tau}$ of the game $\hat{G}$.

    \begin{algorithm}[H]
        \SetAlgoLined
        \SetKwData{Left}{Left}\SetKwData{This}{This}\SetKwData{Up}{Up}
        \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
            
        % print i/p and o/p
        \Input{Weighted Game Arena $G = (V, V_\exists, v_I, E, w)$}
        \Output{Weight function $W^'$}
            
        \Begin{
            \For{$e = (u, v)$ \in $E$}{
                \If{$u$ \in $V \setminus V_\exists$}{
                    $W^{'}(e)$ \longleftarrow  -\infty \\
                }
                
                \Else{
                    $W^{'}(e)$ \longleftarrow max\{cVal$^{v^'}$ \vert (u,$v^'$) \in\; $E$ \;\setminus \{e\}\} \\
                }
            } 
        } \\
    
    \caption{ ConstructWprime: Construct $W^'$ that represents the best value obtainable for a strategy of Eve that differs at the given edge}
    \end{algorithm}
    
    \begin{algorithm}[H]
    \SetKwData{Left}{Left}\SetKwData{This}{This}\SetKwData{Up}{Up}
        \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
        
        % print i/p and o/p
        \Input{Weighted Game Arena $G = (V, V_\exists, v_I, E, w)$ and weight function $W'$}
        \Output{$\^{G}$}
        
        \Being{
        
            Initialize $v_0$ \in $\^{V}$ \setminus $\^{V}_\exists$
            
            Initialize $v_1, v_\bot$ \in $\^{V}_\exists $
        
            \For{$b \in Range(W^{'}) \setminus -\infty $}{
                construct $G^b = (V^b, V_\exists^b, v_I^b, E^b ,\^{w})$ \vert{}  $G^b$ is a copy of $G$ with edges $e$ restricted by $W^{'}(e) \leq b$
                \\
                Initialize weight function $\^{w(e)}$ \longleftarrow w(e) - b \; $\forall\; e \in E $ \\
            }
            
            \For{e in $\^{G}$}{
                w($v_0$, $v_0$) = 0  
                
                w($v_\bot$, $v_\bot$) = $-2w_{max} - 1$
                
                w($v_0$ , $v_1$) = 0
                
                \For{$v_I^b$ in $G^b$}{
                    w($v_1$, $v_I^b$)= 0 \\
                }
                  
                
                \For{$v^b$ \in $V^b$ that don't have a successor}{
                
                    w($v^b$, $v_\bot$) = 0 \\
                }
            }
        }
    
    \caption{ConstructGhat: construct $\^{G}$}
    \end{algorithm}

    \begin{algorithm}
        \SetAlgoLined
        \SetKwData{Left}{Left}\SetKwData{This}{This}\SetKwData{Up}{Up}
        \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
        % \KwData{$G=(X,U)$ such that $G^{tc}$ is an order.}
    % \KwResult{{$G’=(X,V)$ with $V\subseteq U$ such that $G’^{tc}$ is an interval order.}
        \Input{An antagonistic game $\^{G}$, $b$, and $Val$ the payoff function \in $\{Sup, Inf, LimInf, LimSup, \overline{MP},\underline{MP}\}$}
        \Output{$Reg, \sigma, \tau $}
        
        \Beign{
        
            $\hat{\sigma}(v_1)$ = $v_I^b$ and avoids $v_{\bot}$ \\  
            $\hat{\tau}(v_o)$ = $v_I$ \\
            
            \For{b \in $Range(W^{'}) \setminus -\infty $}{
                
                \For{$u^b$ \in $V^b$}{
                    \If{$u^b$ \in $V^b$ \setminus ${V}_\exists^b$ }{
                        
                        $\hat{\tau}(u^b)$ = \{$v^b$ \vert \min(\hat{w}($u^b, v^b$)) and ($u^b, v^b$) \in $E^b$ \} \\
                    }
                    
                    \ElseIf{$u^b$ \in ${V}_\exists^b$}{
                        
                    $\hat{\sigma}(u^b)$ = \{$v^b$ \vert \max(\hat{w}($u^b, v^b$)) and ($u^b, v^b$) \in $E^b$ \} \\
                    }
                }
            }
            
            $Reg$ = $Val_{\hat{G}}^v$($\hat{\sigma}, \hat{\tau}$) \\
            $\sigma$ = $\hat{\sigma}$ \setminus v_1 \\
            $\tau$ = $\hat{\tau}$ \setminus v_o \\
        }
    \caption{ComputeAVal: Compute the antagonistic value of $\^{G}$}
    \end{algorithm}
